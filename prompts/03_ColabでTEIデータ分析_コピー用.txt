■ 第3章 TEI から表データを取り出す > Gemini への依頼 (4)

xml_text を lxml の etree でパースし、root 要素を取得して tree という変数に格納してください。
TEI の名前空間（http://www.tei-c.org/ns/1.0）を ns という変数に辞書として保持しておき、
後の XPath クエリで使えるようにしてください。

確認のため、ルート要素のタグ名と、その直下の子要素のタグ名を出力してください。

from lxml import etree

tree = etree.fromstring(xml_text.encode("utf-8"))
ns = {"tei": "http://www.tei-c.org/ns/1.0"}

print(f"ルート要素: {etree.QName(tree).localname}")
print("子要素:")
for child in tree:
    print(f"  - {etree.QName(child).localname}")

import pandas as pd

records = []
for div in tree.findall(".//tei:div[@type='entry']", ns):
    date_el = div.find(".//tei:date", ns)
    if date_el is None or not date_el.get("when"):
        continue
    month = int(date_el.get("when")[5:7])
    for el in div.findall(".//tei:placeName", ns):
        records.append({"place": el.xpath("string(.)").strip(), "month": month})

long_df = pd.DataFrame(records)
pivot = pd.crosstab(long_df["place"], long_df["month"])
pivot = pivot.loc[pivot.sum(axis=1).sort_values(ascending=False).index]
display(pivot.head(10))

from collections import Counter
import re

target_month = "1887-08"
persons_in_month = []

for div in tree.findall(".//tei:div[@type='entry']", ns):
    date_el = div.find(".//tei:date", ns)
    when = (date_el.get("when") or "") if date_el is not None else ""
    if not when.startswith(target_month):
        continue
    for pn in div.findall(".//tei:persName", ns):
        name = re.sub(r"\s+", "", pn.xpath("string(.)"))
        if name:
            persons_in_month.append(name)

top = pd.Series(Counter(persons_in_month)).sort_values(ascending=False)
top.plot(kind="barh", figsize=(10, 5), title=f"{target_month} に登場した人物")
plt.tight_layout()
plt.show()

import networkx as nx
from collections import Counter

target = "榊原杏平"
co_counter = Counter()

for year, tree in trees.items():
    for div in tree.findall(".//tei:div[@type='entry']", ns):
        names = [re.sub(r"\s+", "", el.xpath("string(.)"))
                 for el in div.findall(".//tei:persName", ns)]
        if target not in names:
            continue
        for n in names:
            if n != target and n:
                co_counter[n] += 1

top_co = co_counter.most_common(15)

G = nx.Graph()
G.add_node(target, size=30)
for name, w in top_co:
    G.add_node(name, size=10 + w)
    G.add_edge(target, name, weight=w)

pos = nx.spring_layout(G, k=1.5, seed=42)
plt.figure(figsize=(12, 8))
nx.draw_networkx_nodes(G, pos,
    node_size=[G.nodes[n].get("size", 10) * 30 for n in G.nodes],
    node_color=["crimson" if n == target else "steelblue" for n in G.nodes])
nx.draw_networkx_labels(G, pos, font_family="IPAexGothic", font_size=10)
nx.draw_networkx_edges(G, pos, width=[G[u][v]["weight"] * 0.5 for u, v in G.edges])
plt.title(f"「{target}」との共起人物")
plt.axis("off")
plt.tight_layout()
plt.show()

- 3x3 の行列（1887・1888・1889）をヒートマップで表示、数値も注記

■ 第6章 複数文書を比較する > Gemini への依頼 (9)

from scipy.spatial.distance import cosine
import seaborn as sns

def cos_sim(c1, c2):
    vocab = sorted(set(c1) | set(c2))
    v1 = [c1.get(w, 0) for w in vocab]
    v2 = [c2.get(w, 0) for w in vocab]
    return 1 - cosine(v1, v2)

years = [1887, 1888, 1889]
fig, axes = plt.subplots(1, 2, figsize=(12, 5))

for ax, (label, dfs, col) in zip(axes, [("人名", pers_dfs, "name"),
                                        ("地名", place_dfs, "place")]):
    counters = {y: Counter(dfs[y][col]) for y in years}
    sim = pd.DataFrame(
        [[cos_sim(counters[a], counters[b]) for b in years] for a in years],
        index=years, columns=years)
    sns.heatmap(sim, annot=True, fmt=".3f", cmap="YlOrRd",
                vmin=0.8, vmax=1, ax=ax)
    ax.set_title(f"{label}のコサイン類似度")

plt.tight_layout()
plt.show()
